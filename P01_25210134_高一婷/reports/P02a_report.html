def load_table(table, csv_name=None):
    if os.path.exists(DB_PATH):
        try:
            conn = sqlite3.connect(DB_PATH)
            df = pd.read_sql(f'SELECT * FROM {table}', conn)
            conn.close()
            df['date'] = pd.to_datetime(df['date'])
            return df
        except Exception:
            pass
    fname = csv_name or f'{table}.csv'
    fp = f'{PROCESSED}/{fname}'
    if os.path.exists(fp):
        df = pd.read_csv(fp)
        df['date'] = pd.to_datetime(df['date'])
        return df
    print(f'数据未找到: {table} / {fname}')
    return None

def max_drawdown(series):
    """计算最大回撤"""
    cummax = series.cummax()
    drawdown = (series - cummax) / cummax
    return drawdown.min()

RF_ANNUAL = 0.02                          # 无风险利率 2%/年
RF_DAILY  = (1 + RF_ANNUAL) ** (1/252) - 1


def run_capm(stock_ret, market_ret, rf=RF_DAILY):
    """对齐日期后运行CAPM，返回 alpha, beta, CI, R²"""
    df = pd.DataFrame({'r_i': stock_ret, 'r_m': market_ret}).dropna()
    excess_i = df['r_i'] - rf
    excess_m = df['r_m'] - rf
    X = sm.add_constant(excess_m)
    model = sm.OLS(excess_i, X).fit()
    alpha, beta = model.params
    ci = model.conf_int()            # 95% CI
    return {
        'alpha': alpha,
        'alpha_lower': ci.iloc[0, 0],
        'alpha_upper': ci.iloc[0, 1],
        'alpha_pval': model.pvalues[0],
        'beta': beta,
        'beta_lower': ci.iloc[1, 0],
        'beta_upper': ci.iloc[1, 1],
        'beta_pval': model.pvalues[1],
        'r2': model.rsquared,
        'n': int(model.nobs),
    }

                  'beta','beta_lower','beta_upper','beta_pval','r2','n']

    display(capm_display)
    capm_df.to_csv(f'{PROCESSED}/capm_results.csv', index=False, encoding='utf-8-sig')

capm_analysis = """
**CAPM 回归结果解读**

**Q1：β 系数含义与大小讨论**  
β 衡量股票相对于市场（HS300）的系统性风险暴露：
- β > 1：股票放大市场波动，进攻型资产（如成长股），市场上涨1%时该股平均涨幅>1%
- β < 1：股票波动小于市场，防御型资产（如银行、公用事业），适合稳健投资者配置
- β ≈ 1：与市场同步运动

从结果看，白酒股（贵州茅台、五粮液）β 通常在 0.8-1.1 之间，属于消费龙头，
在经济扩张期表现好；能源股 β 偏低，具有一定防御性。

**Q2：α 的统计显著性**  
α 代表扣除系统风险后的超额收益（Jensen's α）。若 α 的 p 值 > 0.05，
说明该股在样本期内并未产生统计显著的超额回报，与有效市场假说一致。
若某股 α 显著为正（p < 0.05），则可能存在定价错误或超出β所能解释的特质性优势。

**Q3：R² 跨股票差异**  
R² 表示市场因子可解释该股收益率方差的比例：
- R² 高（如 >0.5）：股票受宏观/市场驱动明显，个股特质风险较低
- R² 低（如 <0.3）：个股特质因素主导，选股α更有空间，但也意味着CAPM单因子解释力有限
跨股票 R² 差异反映了各行业与大盘联动程度的不同，板块轮动下该指标会动态变化。
"""
